74 template <typename T>
75 inline static bool save(
76 const std::string& var
77 ) {
78 verify<T>();
79 if (!isreg(var)) return false;
80 std::ofstream dfile(data_file_name, std::ios::binary | std::ofstream::app);
81 if (!dfile.good()) return false;
82
83 try {
84 T tempv = get<T>(var);
85
86 // Write variable name.
87 {int32_t tempi = std::strlen(var.c_str()) + 1;
88 dfile.write(reinterpret_cast<const char*>(&tempi), sizeof(int32_t));
89 dfile.write(var.c_str(), tempi);}
90
91 // Write variable value.
92 {int32_t tempi;
93 if constexpr (std::is_same_v<std::string, T>)
94 tempi = std::strlen(std::any_cast<const std::string>(tempv).c_str()) + 1;
95 else tempi = sizeof(T);
96 dfile.write(reinterpret_cast<const char*>(&tempi), sizeof(int32_t));
97 if constexpr (std::is_same_v<std::string, T>) // Variable is a string.
98 dfile.write(std::any_cast<const std::string>(tempv).c_str(), tempi);
99 else // Handle simple variables.
100 dfile.write(reinterpret_cast<const char*>(&tempv), tempi);}
101 } catch(...) {
102 dfile.close();
103 return false;
104 }
105
106 dfile.close();
107 return true;

116 template <typename T>
117 inline static bool load(
118 const std::string& var
119 ) {
120 verify<T>();
121 if (!isreg(var)) return false;
122 std::ifstream dfile(data_file_name, std::ios::binary);
123 if (!dfile.good()) return false;
124 dfile.seekg(0, dfile.beg);
125
126 while (true) { // Check entire file.
127 try {
128 if (dfile.peek() == EOF) return false; // Hit end of file and not found.
129 std::string in_var;
130
131 // Read in variable name.
132 {int32_t size;
133 dfile.read(reinterpret_cast<char*>(&size), sizeof(int32_t));
134 char* buffer = new char[size];
135 dfile.read(buffer, size);
136 in_var = std::string(buffer);
137 delete[] buffer;}
138
139 // Read in variable size.
140 int32_t size;
141 dfile.read(reinterpret_cast<char*>(&size), sizeof(int32_t));
142 if (var == in_var) { // Found variable.
143 if constexpr (std::is_same_v<std::string, T>) {
144 // Variable is a string, read accordingly.
145 char* buffer = new char[size];
146 dfile.read(buffer, size);
147 std::string in_val = std::string(buffer);
148 delete[] buffer;
149 set<std::string>(var, in_val);
150 } else {
151 // Handle other data types.
152 T in_val;
153 dfile.read(reinterpret_cast<char*>(&in_val), size);
154 set<T>(var, in_val);
155 }
156 break; // Found value, break loop.
157 }
158 dfile.seekg(size, dfile.cur); // Skip if not correct variable.
159 } catch(...) { // Error reading, fail.
160 dfile.close();
161 return false;
162 }
163 }
164
165 dfile.close();
166 return true;
167 };
